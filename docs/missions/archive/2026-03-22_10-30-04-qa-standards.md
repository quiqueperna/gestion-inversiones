describe('closeTradeUnitWithQuantity', () => {
    const CSV_TWO_BUYS = `...`; // 2 BUY de 10 AAPL a 100, misma account+broker

    beforeEach(() => {
        resetMemoryState();
        initializeMemoryState(CSV_TWO_BUYS, false);
    });

    test('cierre exacto (qty = qty de la TU seleccionada)', async () => {
        const result = await closeTradeUnitWithQuantity({
            symbol: 'AAPL', closeDate: '2024-06-01', closePrice: 110,
            totalQty: 10, broker: 'AMR', account: 'USA',
            primaryEntryExecId: 1
        });
        expect(result.success).toBe(true);
        const state = getMemoryState();
        const closedTUs = state.tradeUnits.filter(tu => tu.status === 'CLOSED');
        expect(closedTUs.length).toBe(1);
        expect(closedTUs[0].pnlNominal).toBeCloseTo(100); // (110-100)*10
    });

    test('cierre parcial (qty < qty de la TU)', async () => {
        await closeTradeUnitWithQuantity({
            symbol: 'AAPL', closeDate: '2024-06-01', closePrice: 110,
            totalQty: 5, broker: 'AMR', account: 'USA',
            primaryEntryExecId: 1
        });
        const state = getMemoryState();
        const openTUs = state.tradeUnits.filter(tu => tu.status === 'OPEN');
        const closedTUs = state.tradeUnits.filter(tu => tu.status === 'CLOSED');
        // Debe haber splitting: 1 cerrada (qty=5) + 1 abierta (qty=5 remainder) + 1 abierta original
        expect(closedTUs.length).toBe(1);
        expect(openTUs.length).toBeGreaterThanOrEqual(2);
        expect(closedTUs[0].qty).toBe(5);
    });

    test('cierre en cascada (qty > qty de TU primaria, cierra múltiples)', async () => {
        await closeTradeUnitWithQuantity({
            symbol: 'AAPL', closeDate: '2024-06-01', closePrice: 110,
            totalQty: 15, broker: 'AMR', account: 'USA',
            primaryEntryExecId: 1
        });
        const state = getMemoryState();
        const closedTUs = state.tradeUnits.filter(tu => tu.status === 'CLOSED');
        expect(closedTUs.length).toBeGreaterThanOrEqual(2);
        const totalClosedQty = closedTUs.reduce((sum, tu) => sum + tu.qty, 0);
        expect(totalClosedQty).toBe(15);
    });

    test('exceso sin contraparte crea Execution SELL abierta', async () => {
        await closeTradeUnitWithQuantity({
            symbol: 'AAPL', closeDate: '2024-06-01', closePrice: 110,
            totalQty: 25, broker: 'AMR', account: 'USA', // más que las 20 disponibles
            primaryEntryExecId: 1
        });
        const state = getMemoryState();
        const openSell = state.executions.find(e =>
            e.symbol === 'AAPL' && e.side === 'SELL' && !e.isClosed
        );
        expect(openSell).toBeDefined();
        expect(openSell?.remainingQty).toBe(5); // 25 - 20 = 5 exceso
    });

describe('initializeMemoryState — aislamiento account+broker', () => {
    test('BUY en USA y SELL en Argentina del mismo símbolo NO se matchean', () => {
        const CSV = `symbol,qty,price,side,account,broker,date,category
AAPL,10,100,BUY,USA,AMR,2024-01-10,TRADE
AAPL,10,110,SELL,Argentina,AMR,2024-02-10,TRADE`;
        resetMemoryState();
        initializeMemoryState(CSV, false);
        const state = getMemoryState();
        // Debe haber 0 TUs cerradas (no matchearon por account diferente)
        const closedTUs = state.tradeUnits.filter(tu => tu.status === 'CLOSED');
        expect(closedTUs.length).toBe(0);
        // Ambas posiciones quedan abiertas
        const openTUs = state.tradeUnits.filter(tu => tu.status === 'OPEN');
        expect(openTUs.length).toBe(2);

    test('BUY en Schwab y SELL en IOL del mismo símbolo NO se matchean', () => {

AAPL,10,110,SELL,USA,IOL,2024-02-10,TRADE`;

## ERRORES A EVITAR (de lessons.md)

1. **`resetMemoryState()` en `beforeEach`** — obligatorio en todos los tests de integración. Sin esto, los tests contaminan el estado entre runs. (lessons.md v7)
2. **Al agregar nueva entidad a memoryState, actualizar `resetMemoryState()`** — el nuevo estado tiene `executions` y `tradeUnits`, asegurarse de que ambos son arrays vacíos en reset. (lessons.md v7)
3. **Tests que cuentan `.length` de arrays** — al cambiar la arquitectura (e.g., open TUs ahora están en `tradeUnits`), todos los tests que hacen `expect(state.tradeUnits.length).toBe(N)` deben revisarse. (lessons.md v4 Error 2)
4. **No usar `--no-verify` en git** — si el pre-commit hook falla, investigar la causa, no bypassear.
5. **`git add` por archivos específicos** — no `git add .`. Revisar `git status` antes del commit. (lessons.md v8 Error 3)
