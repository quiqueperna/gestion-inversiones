### T4 — Agregar tests para Cuentas CRUD
**Archivo:** `src/server/actions/__tests__/cuentas.test.ts` (nuevo)

```typescript
import { beforeEach, describe, it, expect } from 'vitest';
import { resetMemoryState, initializeMemoryState } from '@/lib/data-loader';
import { getCuentas, addMemoryCuenta, removeMemoryCuenta } from '@/server/actions/transactions';

const EMPTY_CSV = `date,symbol,quantity,price,broker,cuenta,type,isFalopa`;

describe('Cuentas CRUD', () => {
    beforeEach(() => {
        resetMemoryState();
        initializeMemoryState(EMPTY_CSV);
    });

    it('retorna cuentas por defecto', async () => {
        const cuentas = await getCuentas();
        expect(cuentas).toHaveLength(3);
        expect(cuentas.map(c => c.nombre)).toContain('USA');
    });

    it('agrega una nueva cuenta', async () => {
        await addMemoryCuenta('Europa', 'Broker europeo');
        const cuentas = await getCuentas();
        expect(cuentas.find(c => c.nombre === 'Europa')).toBeDefined();
    });

    it('elimina una cuenta', async () => {
        const cuentas = await getCuentas();
        const usaId = cuentas.find(c => c.nombre === 'USA')!.id;
        await removeMemoryCuenta(usaId);
        const updated = await getCuentas();
        expect(updated.find(c => c.nombre === 'USA')).toBeUndefined();
    });
});
```

import { test, expect } from '@playwright/test';

test.describe('Trades', () => {
    test.beforeEach(async ({ page }) => {
        await page.goto('/');
        await page.getByRole('button', { name: /trades/i }).click();
        await page.waitForLoadState('networkidle');
    });

        // Seleccionar filtro "Abiertos"
        await page.getByRole('combobox', { name: /estado/i }).selectOption('open');
        // Verificar que todos los trades visibles tienen estado Abierto

        await expect(estadoCells.first()).toBeVisible();
    });

    test('filtra trades por estado Cerrados', async ({ page }) => {
        await page.getByRole('combobox', { name: /estado/i }).selectOption('closed');

        await expect(estadoCells.first()).toBeVisible();

        // Seleccionar filtro de instrumento
        const instrFilter = page.locator('select').filter({ hasText: /stock|instrumento/i });
        await instrFilter.selectOption('STOCK');
        await page.waitForLoadState('networkidle');
        // La tabla debe mostrar resultados (no estar vacía)
        await expect(page.locator('tbody tr').first()).toBeVisible();
